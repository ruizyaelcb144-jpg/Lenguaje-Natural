import re

import nltk
#descarga de modelo de tokenizacion
#nltk.download('punkt_tab')
#importamos las funciones de la libreria
from nltk.tokenize import word_tokenize,sent_tokenize

#ejemplo de texto a tokrnizar
texto=("En la materia de lenguaje natural PLN y vision artificial VA, se desarrolla. "
       "un portafolio de practicas como material de apoyo para los siguientes. "
       "semestres de la especialidad, incluyendo practcas y glosario de funciones. "
       "por cada pr√°ctica. ")

#como buena practica antes de tokenizar,hay que limpiar el texto
#convertir en minuscula
texto_limpio=texto.lower()

#quitar saltos de linea y espacios
texto_limpio=texto_limpio.strip()

#eliminar caracteres que no sean letras
texto_limpio = re.sub(r'[^a-z√°√©√≠√≥√∫√º√±\s]', '', texto_limpio)

#TOKENIZAR
#token por palabras
palabras=word_tokenize(texto_limpio,language='spanish')
print("Token en palabras: ",palabras)

#token por oraciones
oraciones=sent_tokenize(texto)
print("Token en oraciones: ",oraciones)


print("Texto original: ",texto)
print("Texto limpio: ",texto_limpio)
