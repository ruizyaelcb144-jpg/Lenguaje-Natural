from textblob import TextBlob
import random
from deep_translator import GoogleTranslator

usuario=["hola",
         "hola chat"]

preguntas=["¬øcomo te sientes hoy?",
           "¬øque tal estas hoy?",
           ]
respuestas=["me siento triste",
            "estoy muy feliz",
            "estoy enojado",
            "estoy bien"]
positivo=["me alegra que te sientas feliz hoy",
          "ten un excelente dia"]
negativo=["lamento que te sientas triste hoy",
          "lamento mucho lo que sientes mis mejores deseos",
          "amigo no estes enojado"]
neutro=["okey escelente",
        "bien echale ganas"]
while True:
    texto = input("Ingresa una texto: ")
    ingles = GoogleTranslator(source='es', target='en').translate(texto)
    analizar = TextBlob(ingles)
    sentimiento = analizar.sentiment.polarity

    if texto in usuario:
        print("Chatbot: ",random.choice(preguntas))
    if texto in respuestas:
        if sentimiento >= 0.5:
            print("Chatbot: ",random.choice(positivo))
        if sentimiento ==0.0:
            print("Chatbot: ",random.choice(neutro))
        if sentimiento <0.0:
            print("Chatbot: ",random.choice(negativo))
